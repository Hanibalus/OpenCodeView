int noofins(char str[],int n)
{
 int table[n][n]={0},i,l,h;
 
 for(i=1;i<n;i++)
 {
  for(l=0,h=i; h<n; ++l,++h)
  {
   if(str[l]==str[h])
   table[l][h]=table[l+1][h-1];
   
   else
   table[l][h]= min(table[l+1][h], table[l][h-1]) + 1;
  }
 }
 
 return table[0][n-1];
}
