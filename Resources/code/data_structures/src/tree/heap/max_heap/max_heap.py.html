def left(i):
    return 2 * i + 1


def right(i):
    return 2 * i + 2


def heapify(a, i):
    l = left(i)
    r = right(i)
    if l < len(a) and a[l] > a[i]:
        largest = l
    else:
        largest = i
    if r < len(a) and a[r] > a[largest]:
        largest = r
    if largest != i:
        a[i], a[largest] = a[largest], a[i]
        heapify(a, largest)


def build_max_heap(a):
    for i in range((len(a) - 1) / 2, -1, -1):
        heapify(a, i)
