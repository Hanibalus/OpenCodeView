const ll N = 100009;
int n,i,j,sz;
string a;
ll next_[27][N],end_[N];
void add(string a){
    int v = 0;
    for(int i = 0;i < (int)a.size();i++){
       int c = a[i] - 'a';
       if(next_[c][v] == -1)
         v = next_[c][v] = ++sz;
       else
         v = next_[c][v];
    }
    ++end_[v];
}
bool search(string a){
    int v = 0;
    for(i = 0;i < (int)a.size();i++){
       int c = a[i] - 'a';
       if(next_[c][v] == -1) return false;
       v = next_[c][v];
    }
    return end_[v] > 0;
}
